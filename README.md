Windows:

javac -cp ..\..\lib\jgroups-3.6.4.Final.jar Ebanco_visao.java

java "-Djava.net.preferIPv4Stack=true" -cp "..\..\lib\jgroups-3.6.4.Final.jar;..\" pacote.Classe_principal_compilada

javac -cp ../../lib/jgroups-3.6.4.Final.jar Ebanco_visao.java

java -Djava.net.preferIPv4Stack=true -cp ../:../../lib/jgroups-3.6.4.Final.jar pacote.Classe_principal_compilada
